C) AUTH — SIGNUP, LOGIN, REFRESH, LOGOUT
================================================================================

  # --- Sign up (returns profile only; no access token — log in next) ---
  curl -sS -X POST "${API_BASE}/v1/auth/signup" \
    -H "Content-Type: application/json" \
    -d '{"username":"Dev User","email":"YOUR_EMAIL@example.com","password":"YourPassMin8chars"}'

  # --- Login (JSON: access_token + profile; refresh token = httpOnly cookie on /v1/auth) ---
  curl -sS -c /tmp/factora_cookies.txt -X POST "${API_BASE}/v1/auth/login" \
    -H "Content-Type: application/json" \
    -d '{"email":"demo-dashboard@example.org","password":"FactoraDemo2026!"}'

  # Capture access token (needs jq):
  export ACCESS_TOKEN="$(curl -sS -c /tmp/factora_cookies.txt -b /tmp/factora_cookies.txt \
    -X POST "${API_BASE}/v1/auth/login" \
    -H "Content-Type: application/json" \
    -d '{"email":"demo-dashboard@example.org","password":"FactoraDemo2026!"}' \
    | jq -r '.access_token')"

  echo "$ACCESS_TOKEN"

  # --- Refresh (sends cookie automatically; returns new access_token + profile) ---
  curl -sS -c /tmp/factora_cookies.txt -b /tmp/factora_cookies.txt \
    -X POST "${API_BASE}/v1/auth/refresh" \
    -H "Content-Type: application/json" \
    -d '{}'

  # --- Logout (revokes refresh session; clears cookie) ---
  curl -sS -c /tmp/factora_cookies.txt -b /tmp/factora_cookies.txt \
    -X POST "${API_BASE}/v1/auth/logout" \
    -H "Content-Type: application/json" \
    -d '{}'

D) AUTH — FORGOT PASSWORD & RESET PASSWORD (FULL WORKFLOW)
================================================================================

  Notes:
  - Forgot-password requires the account EMAIL (not username). If you forgot it,
    look up in DB: SELECT email, username FROM users WHERE username ILIKE '%plilip%';
  - API always returns the same generic success message (anti-enumeration).
  - Reset link in email: {FRONTEND_BASE_URL}/auth/reset-password?token=...
    Copy the raw "token" query value for the API below if you skip the UI.
  - Email delivery uses NotificationService (Brevo); check logs if nothing arrives.

  # --- Step 1 — request reset email ---
  curl -sS -X POST "${API_BASE}/v1/auth/forgot-password" \
    -H "Content-Type: application/json" \
    -d '{"email":"YOUR_EMAIL@example.com"}'

  # Expected: 200 + message like "If an account exists for that email, a reset link has been sent."

  # --- Step 2 — open email, copy token from link, or use frontend reset page ---

  # --- Step 3 — set new password (token expires in ~1 hour) ---
  curl -sS -X POST "${API_BASE}/v1/auth/reset-password" \
    -H "Content-Type: application/json" \
    -d '{
      "token":"PASTE_TOKEN_FROM_EMAIL_OR_URL",
      "new_password":"YourNewPasswordMin8",
      "confirm_password":"YourNewPasswordMin8"
    }'

  # --- Step 4 — log in with the new password ---
  curl -sS -c /tmp/factora_cookies.txt -X POST "${API_BASE}/v1/auth/login" \
    -H "Content-Type: application/json" \
    -d '{"email":"YOUR_EMAIL@example.com","password":"YourNewPasswordMin8"}'

E) AUTH — CHANGE PASSWORD (LOGGED IN)
================================================================================

  # Revokes all sessions after success; you must log in again with new password.

  curl -sS -X POST "${API_BASE}/v1/auth/change-password" \
    -H "Authorization: Bearer ${ACCESS_TOKEN}" \
    -H "Content-Type: application/json" \
    -d '{
      "current_password":"YourOldPassword",
      "new_password":"YourNewPasswordMin8",
      "confirm_password":"YourNewPasswordMin8"
    }'
